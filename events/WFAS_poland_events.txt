#Confederation of Bar
country_event = {
	id = WFAS_poland.4
	title = "WFAS_poland_bar_title"
	desc = "WFAS_poland_bar_desc"
	picture = LIBERUM_VETO_eventPicture
	fire_only_once = yes
	trigger = {
		tag = PLC
		NOT = {
			war_with = RUS
		}
		NOT = {
			alliance_with = RUS
		}
		RUS = {
			exists = yes
			army_strength = {
				who = ROOT
				value = 1.5
			}
		}
		is_year = 1750
	}
	mean_time_to_happen = {
		months = 120
	}
	option = {
		name = "WFAS_poland_bar_option"
		IF = {
			limit = {
				owns = 281
			}
			281 = {
				spawn_rebels = {
					size = 3
					type = polish_noble_rebels
				}
			}
		}
		ELSE = {
			random_owned_province = {
				limit = {
					continent = europe
				}
				spawn_rebels = {
					size = 3
					type = polish_noble_rebels
				}
			}
		}
		reverse_add_casus_belli = {
			type = cb_vassalize_mission
			target = RUS
		}
		set_country_flag = WFAS_poland_partition_start
	}
}

#Duchy of Estonia
country_event = {
	id = WFAS_poland.6
	title = "WFAS_poland_estonia_title"
	desc = "WFAS_poland_estonia_desc"
	fire_only_once = yes
	trigger = {
		OR = {
			tag = PLC
			tag = LIT
			tag = POL
		}
		any_owned_province = {
			area = estonia_ingria_area
		}
		NOT = {
			any_owned_province = {
				area = estonia_ingria_area
				any_neighbor_province = {
					owned_by = ROOT
				}
			}
		}
		NOT = {
			exists = EST
		}
		is_at_war = no
		is_year = 1500
	}
	mean_time_to_happen = {
		months = 60
	}
	immediate = {
		hidden_effect = {
			estonia_ingria_area = {
				add_core = EST
			}
		}
	}
	option = {
		name = "WFAS_poland_estonia_autonomy"
		estonia_ingria_area = {
			limit = {
				owned_by = ROOT
			}
			add_province_modifier = {
				name = WFAS_poland_estonia_autonomy_modifier
				duration = -1
			}
		}
	}
	option = {
		name = "WFAS_poland_estonia_vassal"
		release = EST
		create_subject = {
			subject_type = vassal
			subject = EST
		}
		EST = {
			add_opinion = {
				who = ROOT
				modifier = released_vassal
			}
			hidden_effect = {
				define_ruler = {
					name = "Jaagup"
					dynasty = "Fridrich"
					male = yes
				}
			}
		}
	}
	option = {
		name = "WFAS_poland_estonia_kurland"
		trigger = {
			KUR = {
				exists = yes
				is_subject_of = ROOT
				any_owned_province = {
					any_neighbor_province = {
						area = estonia_ingria_area
					}
				}
			}
		}
		estonia_ingria_area = {
			limit = {
				owned_by = ROOT
			}
			cede_province = KUR
		}
		KUR = {
			add_opinion = {
				who = ROOT
				modifier = opinion_pleased
			}
			add_liberty_desire = -25
		}
	}
}
