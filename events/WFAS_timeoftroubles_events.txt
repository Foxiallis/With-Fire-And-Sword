	id = WFAS_timeoftroubles.1
	title = "WFAS_tot_son_title"
	desc = "WFAS_tot_son_desc"
	picture = BAD_WITH_MONARCH_eventPicture
	fire_only_once = yes
	trigger = {
		OR = {
			tag = RUS
			AND = {
				tag = MOS
				NOT = {
					exists = RUS
				}
			}
		}
		dynasty = "Rurikovich"
		heir_dynasty = "Rurikovich"
		has_heir = yes
		is_year = 1575
	}
	mean_time_to_happen = {
		months = 72
	}
	immediate = {
		hidden_effect = {
			set_country_flag = WFAS_tot_killed_son
			add_country_modifier = {
				name = WFAS_tot_prelude
				hidden = yes
				duration = -1
			}
		}
	}
	option = {
		name = "WFAS_tot_son_we"
		kill_heir = yes
		add_war_exhaustion = -5
		ai_chance = {
			factor = 0
			modifier = {
				factor = 1
				war_exhaustion = 4
			}
		}
	}
	option = {
		name = "WFAS_tot_son_church"
		kill_heir = yes
		IF = {
			limit = {
				NOT = {
					capital_scope = {
						OR = {
							has_building = temple
							has_building = cathedral
						}
					}
				}
			}
			capital_scope = {
				add_building = temple
			}
		}
		ELSE = {
			random_owned_province = {
				limit = {
					NOT = {
						OR = {
							has_building = temple
							has_building = cathedral
						}
					}
				}
				add_building = temple
			}
		}
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				NOT = {
					war_exhaustion = 4
				}
			}
		}
	}
}
